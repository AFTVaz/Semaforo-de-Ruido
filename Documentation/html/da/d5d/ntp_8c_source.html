34static bool NTP_ParseResponse( const char *buffer, void *user_data ){
35 if (buffer == NULL || user_data == NULL) return false;
36
37 const char *ipd = strstr(buffer, "+IPD,");
38 if (ipd) {
39 int link_id = -1;
40 int data_len = 0;
41 const char *header_end = strchr(ipd, ':');
42
43 if (header_end){
44 int fields = sscanf(ipd, "+IPD,%d,%d:", &link_id, &data_len);
45 if (fields == 1) {
46 data_len = link_id;
47 link_id = -1;
48 } else if (fields == 2) {
49 if(link_id != NTP_SERVICE_ID) return false;
50 } else return false;
51
52 if ( data_len == NTP_PACKET_SIZE ){
53 NTP_Packet rxPacket;
54
55 int len = ESPSERIAL_Recv((unsigned char *)&rxPacket, NTP_PACKET_SIZE);
56 if ( len == NTP_PACKET_SIZE ){
57 uint32_t secs_net = rxPacket.txTm_s;
58 uint32_t secs = ntohl(secs_net);
59
60 time_t *timestamp = (time_t *)user_data;
61 *timestamp = (time_t)(secs - UNIX_EPOCH);
62 return true;
63 }
64 }
65 }
66 }
67 return false;
68}

70static bool NTP_Start( void ){
71 WIFI_ResultValueType result;
72 char cmd[128];
73
74 int cipmux = WIFI_GetCIPMUX();
75
76 if (cipmux == 0) {
77 snprintf(cmd, sizeof(cmd), "AT+CIPSTART=\"%s\",\"%s\",%d\r\n",
78 NTP_CONN_TYPE, NTP_SERVER, NTP_REMOTE_PORT);
79 } else {
80 snprintf(cmd, sizeof(cmd), "AT+CIPSTART=%d,\"%s\",\"%s\",%d\r\n",
81 NTP_SERVICE_ID, NTP_CONN_TYPE, NTP_SERVER, NTP_REMOTE_PORT);
82 }
83 if(!WIFI_RequestExpected(cmd, "OK\r\n", NTP_CIPSTART_TIMEOUT)) return false;
84 while ((result = WIFI_GetResult()) == WIFI_RESULT_NONE) taskYIELD();
85 return result == WIFI_RESULT_SUCCESS;

88static bool NTP_SendAndReceive( time_t *timestamp ) {
89 if (timestamp == NULL) return false;
90
91 WIFI_ResultValueType result;
92 char cmd[128];
93 NTP_Packet packet;
94 memset(&packet, 0, sizeof(NTP_Packet));
95
96 packet.li_vn_mode = 0xE3; // LI=3, VN=4, Mode=3
97 packet.stratum = 0; // Stratum
98 packet.poll = 6; // Poll interval
99 packet.precision = 0xEC; // Precision
100
101 int cipmux = WIFI_GetCIPMUX();
102
103 if (cipmux == 0) {
104 snprintf( cmd, sizeof(cmd), "AT+CIPSEND=%d\r\n", NTP_PACKET_SIZE );
105 } else {
106 snprintf( cmd, sizeof(cmd), "AT+CIPSEND=%d,%d\r\n", NTP_SERVICE_ID, NTP_PACKET_SIZE );
107 }
108
109 TickType_t xStart = xTaskGetTickCount();
110 while ((xTaskGetTickCount() - xStart) < NTP_TRANSACTION_TIMEOUT){
111 if(WIFI_RequestPacketTransaction(
112 cmd,
113 (uint8_t*)&packet,
114 NTP_PACKET_SIZE,
115 NTP_TRANSACTION_TIMEOUT,
116 NTP_ParseResponse,
117 timestamp)
118 ){
119 while ((result = WIFI_GetResult()) == WIFI_RESULT_NONE) taskYIELD();
120 return result == WIFI_RESULT_SUCCESS;
121 }
122 else vTaskDelay(NTP_WAIT_DELAY);
123 }
124
125 return false;
126}

155static void NTP_Task(void *pvParameters) {
156
157 bool connStart = false;
158
159 time_t timestamp = 0;
160
161 while (true) {
162
163 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
164
165 if (WIFI_GetState() == WIFI_STATE_CONNECTED) {
166 if (NTP_GetState() == NTP_PROCESSING) {
167 if(!connStart) connStart = NTP_Start();
168
169 if(connStart){
170 if( NTP_SendAndReceive( &timestamp) ){
171 if ( xSemaphoreTake(xNTPMutex, portMAX_DELAY) == pdTRUE ){
172 ntp_timestamp = timestamp;
173 xNTPState = NTP_READY;
174 xSemaphoreGive(xNTPMutex);
175 }
176 } else {
177 NTP_SetState( NTP_IDLE );
178 NTP_Close();
179 connStart = false;
180 }
181 }
182 else{
183 NTP_SetState( NTP_IDLE );
184 }
185 }
186 } else {
187 if (NTP_GetState() != NTP_IDLE) {
188 NTP_SetState(NTP_IDLE);
189 }
190 }
191 }

224bool NTP_GetTimestamp( time_t *timestamp ){
225 if (timestamp == NULL) return false;
226
227 bool success = false;
228
229 if (xSemaphoreTake(xNTPMutex, portMAX_DELAY) == pdTRUE) {
230 if (xNTPState == NTP_READY) {
231 time_t t = ntp_timestamp;
232 if (t != 0) {
233 *timestamp = t;
234 success = true;
235 }
236 xNTPState = NTP_IDLE;
237 }
238 xSemaphoreGive(xNTPMutex);
239 }
240 return success;
241}

243bool NTP_RequestAndWait( time_t *timestamp ){
244
245 TickType_t xStart = xTaskGetTickCount();
246
247 NTP_RequestTime();
248 while ((xTaskGetTickCount() - xStart) < NTP_TRANSACTION_TIMEOUT){
249 if (NTP_GetTimestamp( timestamp )) {
250 return true;
251 }
252
253 if (NTP_GetState() == NTP_IDLE) {
254 break;
255 }
256
257 vTaskDelay(NTP_WAIT_DELAY);
258 }
259
260 return false;
261}
