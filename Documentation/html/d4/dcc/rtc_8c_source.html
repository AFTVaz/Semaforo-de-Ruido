37void RTC_GetTimeDate( struct tm *dateTime ){
38 uint32_t CTIME0 = LPC_RTC->CTIME0;
39 uint32_t CTIME1 = LPC_RTC->CTIME1;
40 uint32_t CTIME2 = LPC_RTC->CTIME2;
41
42 dateTime->tm_sec = (CTIME_SEC_MASK & CTIME0);
43 dateTime->tm_min = (CTIME_MIN_MASK & CTIME0) >> 8;
44 dateTime->tm_hour = (CTIME_HOUR_MASK & CTIME0) >> 16;
45 dateTime->tm_wday = (CTIME_DOW_MASK & CTIME0) >> 24;
46 dateTime->tm_mday = (CTIME_DOM_MASK & CTIME1);
47 dateTime->tm_mon = ((CTIME_MONTH_MASK & CTIME1) >> 8) - 1;
48 dateTime->tm_year = ((CTIME_YEAR_MASK & CTIME1) >> 16) - 1900;
49 dateTime->tm_yday = (CTIME_DOY_MASK & CTIME2) - 1;
50 dateTime->tm_isdst = 0; //Daylight Saving is not in effect
51}

53void RTC_SetTimeDate( struct tm *dateTime ){
54 struct tm testTime;
55 RTC_GetTimeDate( &testTime );
56
57 while(compareByTime(&testTime, dateTime)){
58
59 LPC_RTC->CCR = ~( 1 << CLKEN_BIT_POS); //Clock disable
60
61 LPC_RTC->SEC = dateTime->tm_sec;
62 LPC_RTC->MIN = dateTime->tm_min;
63 LPC_RTC->HOUR = dateTime->tm_hour;
64 LPC_RTC->DOM = dateTime->tm_mday;
65 LPC_RTC->DOW = dateTime->tm_wday;
66 LPC_RTC->DOY = dateTime->tm_yday + 1;
67 LPC_RTC->MONTH = dateTime->tm_mon + 1;
68 LPC_RTC->YEAR = dateTime->tm_year + 1900;
69
70 LPC_RTC->CCR = (1 << CLKEN_BIT_POS); //Clock enable
71
72 RTC_GetTimeDate( &testTime );
73 }
74}
