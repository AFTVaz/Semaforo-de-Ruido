22static unsigned int FLASH_GetSectorFromAddress(unsigned int address) {
23 if (address < 0x00010000) {
24 // 4KB sectors (0x0000 0000 to 0x0000 FFFF)
25 return address / 0x1000; // 4 KB = 0x1000
26 } else if (address < 0x00080000) {
27 // 32KB sectors (0x0001 0000 to 0x0007 FFFF)
28 return 16 + ((address - 0x00010000) / 0x8000); // 32 KB = 0x8000
29 } else {
30 return 0xFFFFFFFF; // Invalid address
31 }
32}
