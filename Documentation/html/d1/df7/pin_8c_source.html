21void PIN_Direction(PIN_ValueType pinId, PIN_DIRType dir)
22{
23 uint32_t port = pinId / LPC_GPIO_NUM_PINS;
24 uint32_t pin = pinId % LPC_GPIO_NUM_PINS;
25 LPC_GPIO_TypeDef *gpio = (LPC_GPIO_TypeDef *)(LPC_GPIO_BASE + (port * LPC_GPIO_SIZE));
26
27 if( dir == OUTPUT ){
28 gpio->FIODIR |= (0x1UL << pin);
29 }
30 else{
31 gpio->FIODIR &= ~(0x1UL << pin);
32 }
33}
