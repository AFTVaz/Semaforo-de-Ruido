20#define MAX9814_NUM_SAMPLES 50
21#define MAX9814_BIAS_MIN 1.14f
22#define MAX9814_BIAS_MAX 1.32f

33static uint16_t mic_buffer[MAX9814_NUM_SAMPLES];
34static uint8_t mic_channel = 0;

36static float adc_offset = 2048.0f;

39static float mic_ref_Vout = 0.0f;

42static float mic_bias_min = MAX9814_BIAS_MIN/ADC_RESOLUTION;

45static void MAX9814_Calculate_Sensitivity( void ) {
46 float mic_sensitivity_v = powf(10.0f, MAX9814_SENSITIVITY / 20.0f);
47 mic_ref_Vout = mic_sensitivity_v * powf(10.0f, mic_gain_db / 20.0f);
48}

50static float MAX9814_Calibrate_ADC_Offset(uint8_t channel, uint32_t samples) {
51 uint64_t sum = 0;
52 for (uint32_t i = 0; i < samples; i++) {
53 sum += ADC_Read(channel);
54 }
55 float offset_adc = (float)sum / (float)samples;
56
57 if (offset_adc < mic_bias_min)
58 offset_adc = mic_bias_min;
59 else if (offset_adc > mic_bias_max)
60 offset_adc = mic_bias_max;
61
62 return offset_adc;
63}

65static float MAX9814_Calculate_ADC_Average( uint16_t *samples, uint32_t len, float offset ){
66 float sum_squares = 0.0f;
67 for (uint32_t i = 0; i < len; i++) {
68 float centered = (float)samples[i] - offset;
69 sum_squares += centered * centered;
70 }
71 return sqrtf(sum_squares / (float)len);
72}

74static float MAX9814_Conversion_To_DB( float adc_avg_value ){
75 float v_out = adc_avg_value * ADC_RESOLUTION;
76 return 20.0f * log10f(v_out / mic_ref_Vout) + MAX9814_DB_SPL_REF;
77}

79void MAX9814_Init( MAX9814_GainValueType mic_gain ){
80 ADC_Init(mic_channel);
81 PINCFG_Function( MAX9814_GAIN_PIN, GPIO );
82 MAX9814_SetGain(mic_gain);
83 adc_offset = MAX9814_Calibrate_ADC_Offset(mic_channel, OFFSET_CAL_SAMPLES);
84}

86float MAX9814_Read(void) {
87 for (uint32_t i = 0; i < MAX9814_NUM_SAMPLES; i++) {
88 mic_buffer[i] = ADC_Read(mic_channel);
89 }
90
91 float adc_avg = MAX9814_Calculate_ADC_Average(mic_buffer, MAX9814_NUM_SAMPLES, adc_offset);
92 return MAX9814_Conversion_To_DB(adc_avg);
93}

95void MAX9814_SetGain( MAX9814_GainValueType gain ){
96 current_gain = gain;
97 switch (gain) {
98 case MAX9814_VCC_GAIN:
99 PIN_Direction(MAX9814_GAIN_PIN, OUTPUT);
100 PIN_Clear(MAX9814_GAIN_PIN);
101 mic_gain_db = 40.0f;
102 MAX9814_Calculate_Sensitivity();
103 break;
104 case MAX9814_GND_GAIN:
105 PIN_Direction(MAX9814_GAIN_PIN, OUTPUT);
106 PIN_Set(MAX9814_GAIN_PIN);
107 mic_gain_db = 50.0f;
108 MAX9814_Calculate_Sensitivity();
109 break;
110 case MAX9814_FLOAT_GAIN:
111 PIN_Direction(MAX9814_GAIN_PIN, INPUT);
112 mic_gain_db = 60.0f;
113 MAX9814_Calculate_Sensitivity();
114 break;
115 }
