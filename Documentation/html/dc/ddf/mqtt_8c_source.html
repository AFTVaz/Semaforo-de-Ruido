61static bool MQTT_ParsePacket( const char *buffer, void *user_data ){
62 if (buffer == NULL || user_data == NULL) return false;
63
64 const char *ipd = strstr(buffer, "+IPD,");
65 if (ipd) {
66 int link_id = -1;
67 int data_len = 0;
68 const char *header_end = strchr(ipd, ':');
69
70 if (header_end){
71 int fields = sscanf(ipd, "+IPD,%d,%d:", &link_id, &data_len);
72 if (fields == 1) {
73 data_len = link_id;
74 link_id = -1;
75 } else if (fields == 2) {
76 if(link_id != MQTT_SERVICE_ID) return false;
77 } else return false;
78
79 char data[data_len];
80
81 int len = ESPSERIAL_Recv((unsigned char *)&data, data_len);
82 if (len == data_len){
83 MQTT_Item *info = (MQTT_Item *)user_data;
84 memcpy(info->mqtt_payload, data, data_len);
85 info->mqtt_available_length = data_len;
86
87 return true;
88 }
89 }
90 }
91 return false;
92}

100static int MQTT_Send( unsigned char *buffer, unsigned int len ){
101
102 if (buffer == NULL || len == 0) return MQTT_ERROR;
103
104 WIFI_ResultValueType result;
105 char cmd[128];
106 int cipmux = WIFI_GetCIPMUX();
107
108 if (cipmux == 0) {
109 snprintf( cmd, sizeof(cmd), "AT+CIPSEND=%d\r\n", len );
110 } else {
111 snprintf( cmd, sizeof(cmd), "AT+CIPSEND=%d,%d\r\n", MQTT_SERVICE_ID, len );
112 }
113
114 if(WIFI_RequestSendPacket( cmd, buffer, len, MQTT_CMD_TIMEOUT )){
115 MQTT_SetStatus(MQTT_PROCESSING);
116 while((result = WIFI_GetResult()) == WIFI_RESULT_NONE) taskYIELD();
117 if(result == WIFI_RESULT_SUCCESS) return len;
118 }
119
120 MQTT_SetStatus(MQTT_IDLE);
121 return MQTT_ERROR;
122}

124static int MQTT_Recv( unsigned char *buffer, unsigned int maxlen ) {
125
126 if (buffer == NULL || maxlen == 0) return 0;
127
128 WIFI_ResultValueType result;
129 int total_cpy = 0;
130
131 if( info.mqtt_available_length == 0 ){
132 if(!WIFI_RequestReceive(MQTT_CMD_TIMEOUT, MQTT_ParsePacket, &info)){
133 MQTT_SetStatus(MQTT_IDLE);
134 return 0;
135 }
136 MQTT_SetStatus(MQTT_IDLE);
137 while((result = WIFI_GetResult()) == WIFI_RESULT_NONE) taskYIELD();
138 if(result == WIFI_RESULT_ERROR){
139 return 0;
140 }
141 }
142
143 total_cpy = (info.mqtt_available_length < maxlen) ? info.mqtt_available_length : maxlen;
144 memcpy(buffer, info.mqtt_payload, total_cpy);
145 memmove(info.mqtt_payload, info.mqtt_payload + total_cpy, info.mqtt_available_length - total_cpy);
146 info.mqtt_available_length -= total_cpy;
147
148 return total_cpy;
149}

151static bool MQTT_Connect( const char *host, const unsigned short int port,
152 const unsigned short int keepalive ){
153 WIFI_ResultValueType result;
154 char cmd[128];
155
156 int cipmux = WIFI_GetCIPMUX();
157
158 if (cipmux == 0) {
159 snprintf(cmd, sizeof(cmd), "AT+CIPSTART=\"TCP\",\"%s\",%d,%d\r\n", host, port, keepalive);
160 } else {
161 snprintf(cmd, sizeof(cmd), "AT+CIPSTART=%d,\"TCP\",\"%s\",%d,%d\r\n", MQTT_SERVICE_ID, host, port, keepalive);
162 }
163 if(!WIFI_RequestExpected(cmd, "CONNECT", WIFI_CONN_SUCESS_TIMEOUT)) return false;
164 while ((result = WIFI_GetResult()) == WIFI_RESULT_NONE) taskYIELD();
165 return result == WIFI_RESULT_SUCCESS;
166}

168static bool MQTT_Disconnect(void) {
169 WIFI_ResultValueType result;
170 char cmd[32];
171
172 int cipmux = WIFI_GetCIPMUX();
173 if (cipmux == 0)
174 snprintf(cmd, sizeof(cmd), "AT+CIPCLOSE\r\n");
175 else
176 snprintf(cmd, sizeof(cmd), "AT+CIPCLOSE=%d\r\n", MQTT_SERVICE_ID);
177
178 while (!WIFI_RequestExpected(cmd, "OK\r\n", MQTT_CIPCLOSE_TIMEOUT))
179 vTaskDelay(MQTT_WAIT_DELAY);
180
181 while ((result = WIFI_GetResult()) == WIFI_RESULT_NONE)
182 taskYIELD();
183
184 return result == WIFI_RESULT_SUCCESS;
185}

187static void MQTT_Task( void *pvParameters ){
188
189 unsigned char buffer[256];
190 MQTTTransport transporter;
191 MQTTString topic = MQTTString_initializer;
192 int result;
193 int length;
194
195 TickType_t xQueueWaitTime = xMQTT_ConfigCallback ? pdMS_TO_TICKS(100) : portMAX_DELAY;
196
197 // ESP8266 Transport Layer
198 static transport_iofunctions_t iof = { MQTT_Send, MQTT_Recv };
199 int transport_socket = transport_open(&iof);
200
201 while (true) {
202
203 if( WIFI_GetState() != WIFI_STATE_CONNECTED){
204 MQTT_SetState(MQTT_STATE_INIT);
205 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
206 }
207
208 switch (xMQTT_State) {
209 case MQTT_STATE_INIT:
210 MQTT_SetStatus(MQTT_IDLE);
211 if ( MQTT_Connect(MQTT_ADDRESS, MQTT_PORT, CONNECTION_KEEPALIVE_S * 2) ){
212 MQTT_SetState(MQTT_STATE_CONNECT);
213 }
214 break;
215
216 case MQTT_STATE_CONNECT:
217 transporter.sck = &transport_socket;
218 transporter.getfn = transport_getdatanb;
219 transporter.state = 0;
220
221 MQTTPacket_connectData connectData = MQTTPacket_connectData_initializer;
222 connectData.MQTTVersion = 3;
223 connectData.username.cstring = MQTT_DEVICE_TOKEN;
224 connectData.clientID.cstring = MQTT_DEVICE_TOKEN;
225 connectData.keepAliveInterval = CONNECTION_KEEPALIVE_S;
226 length = MQTTSerialize_connect(buffer, sizeof(buffer), &connectData);
227 // Send CONNECT to the mqtt broker.
228 if ((result = transport_sendPacketBuffer(transport_socket, buffer, length)) == length){
229 MQTT_SetState(MQTT_STATE_WAIT_CONNECT);
230 } else {
231 MQTT_SetState(MQTT_STATE_INIT);
232 }
233 break;
234
235 case MQTT_STATE_WAIT_CONNECT:
236 // Wait for CONNACK response from the MQTT broker.
237 while (true) {
238 if ((result = MQTTPacket_readnb(buffer, sizeof(buffer), &transporter)) == CONNACK) {
239 // Check if the connection was accepted.
240 unsigned char sessionPresent, connack_rc;
241 if ((MQTTDeserialize_connack(&sessionPresent, &connack_rc, buffer,
242 sizeof(buffer)) != 1) || (connack_rc != 0)) {
243 MQTT_SetState(MQTT_STATE_INIT);
244 break;
245 } else {
246 if(xMQTT_ConfigCallback) MQTT_SetState(MQTT_STATE_SUBSCRIBE);
247 else MQTT_SetState(MQTT_STATE_PUBLISH);
248 break;
249 }
250 } else if (result == MQTTPACKET_READ_ERROR) {
251 MQTT_SetState(MQTT_STATE_INIT);
252 break;
253 }
254 }
255 break;
256
257 case MQTT_STATE_SUBSCRIBE:
258 topic.cstring = "v1/devices/me/attributes";
259
260 unsigned char dup = 0; // sem duplica√ß√£o
261 unsigned short packetId = 1; // identificador do pacote
262 int count = 1; // n√∫mero de t√≥picos
263 int req_qos[1] = {1}; // array de QoS por t√≥pico
264
265 int len = MQTTSerialize_subscribe(buffer, sizeof(buffer), dup, packetId, count, &topic, req_qos);
266
267 if (transport_sendPacketBuffer(transport_socket, buffer, len) == len) {
268 MQTT_SetState(MQTT_STATE_WAIT_SUBACK);
269 } else {
270 MQTT_SetState(MQTT_STATE_INIT);
271 }
272 break;
273
274 case MQTT_STATE_WAIT_SUBACK:
275 while (true) {
276 if((result = MQTTPacket_readnb(buffer, sizeof(buffer), &transporter)) == SUBACK){
277
278 unsigned short returnedPacketId;
279 const int maxGrantedQoS = 1;
280 int countGranted;
281 int grantedQoSs[1];
282
283 if ((MQTTDeserialize_suback(&returnedPacketId, maxGrantedQoS, &countGranted, grantedQoSs, buffer, sizeof(buffer)) != 1)
284 || (returnedPacketId != packetId)
285 || (countGranted != maxGrantedQoS)
286 || (grantedQoSs[0] == MQTT_SUBACK_REJECTED)) {
287 MQTT_SetState(MQTT_STATE_INIT);
288 break;
289 }
290
291 MQTT_SetState(MQTT_STATE_PUBLISH);
292 break;
293 } else if (result == MQTTPACKET_READ_ERROR) {
294 MQTT_SetState(MQTT_STATE_INIT);
295 break;
296 }
297 }
298 break;
299
300 case MQTT_STATE_PUBLISH:
301
302 char *payload = NULL;
303
304 if (xQueueReceive(xMQTTQueue, &payload, xQueueWaitTime)) {
305 MQTTString topic = MQTTString_initializer;
306 topic.cstring = "v1/devices/me/telemetry";
307
308 unsigned char dup = 0;
309 int qos = 0;
310 int retained = 0;
311 unsigned short packetId = 0;
312
313 int length = strlen(payload);
314 length = MQTTSerialize_publish(buffer, sizeof(buffer), dup, qos, retained,
315 packetId, topic, (unsigned char *)payload, length);
316
317 if ((result = transport_sendPacketBuffer(transport_socket, buffer, length)) != length) {
318 MQTT_Disconnect();
319 MQTT_SetState(MQTT_STATE_INIT);
320 break;
321 }
322 }
323
324 if (xMQTT_ConfigCallback){
325 result = MQTTPacket_readnb(buffer, sizeof(buffer), &transporter);
326 if (result == PUBLISH) {
327 unsigned char dup, retained;
328 int qos;
329 unsigned short packet_id;
330 unsigned char *payload_in;
331 int payload_len;
332 MQTTString received_topic;
333
334 if (MQTTDeserialize_publish(&dup, &qos, &retained,
335 &packet_id, &received_topic, &payload_in,
336 &payload_len, buffer, sizeof(buffer))) {
337
338 if (strstr(received_topic.lenstring.data, "attributes")) {
339 MQTT_UpdateAttributes((const char *)payload_in, payload_len);
340 }
341 }
342 } else if (result == MQTTPACKET_READ_ERROR) {
343 MQTT_Disconnect();
344 MQTT_SetState(MQTT_STATE_INIT);
345 break;
346 }
347 }
348
349 break;
350
351 default:
352 MQTT_SetState(MQTT_STATE_INIT);
353 break;
354 }
355 }
