18void ADC_Init( uint8_t channel ) {
19 if(channel > 7) return;
20
21 LPC_SC->PCONP |= (1 << 12);
22 LPC_SC->PCLKSEL0 &= ~(3 << 24);
23
24 switch(channel) {
25 case 0:
26 PINCFG_Function(ADC_AD00_PIN, AD0_F1); // P0.23 como AD0.0
27 PINCFG_Mode(ADC_AD00_PIN, NONE);
28 break;
29 case 1:
30 PINCFG_Function(ADC_AD01_PIN, AD0_F1); // P0.24 como AD0.1
31 PINCFG_Mode(ADC_AD01_PIN, NONE);
32 break;
33 case 2:
34 PINCFG_Function(ADC_AD02_PIN, AD0_F1); // P0.25 como AD0.2
35 PINCFG_Mode(ADC_AD02_PIN, NONE);
36 break;
37 case 3:
38 PINCFG_Function(ADC_AD03_PIN, AD0_F1); // P0.26 como AD0.3
39 PINCFG_Mode(ADC_AD03_PIN, NONE);
40 break;
41 case 4:
42 PINCFG_Function(ADC_AD04_PIN, AD0_F3); // P1.30 como AD0.4
43 PINCFG_Mode(ADC_AD04_PIN, NONE);
44 break;
45 case 5:
46 PINCFG_Function(ADC_AD05_PIN, AD0_F3); // P1.31 como AD0.5
47 PINCFG_Mode(ADC_AD05_PIN, NONE);
48 break;
49 case 6:
50 PINCFG_Function(ADC_AD06_PIN, AD0_F2); // P0.2 como AD0.6
51 PINCFG_Mode(ADC_AD06_PIN, NONE);
52 break;
53 case 7:
54 PINCFG_Function(ADC_AD07_PIN, AD0_F2); // P0.3 como AD0.7
55 PINCFG_Mode(ADC_AD07_PIN, NONE);
56 break;
57 }
58
59 LPC_ADC->ADCR = ( 1 << channel ) | ( 4 << ADCR_CLKDIV ) | ( 1 << ADCR_PDN );
60}

62uint16_t ADC_Read(uint8_t channel) {
63 LPC_ADC->ADCR &= ~( ADCR_CLEAN_SEL );
64 LPC_ADC->ADCR |= ( 1 << channel );
65 LPC_ADC->ADCR |= ( 1 << ADCR_START );
66
67 while (!(LPC_ADC->ADGDR & (1 << 31)));
68
69 LPC_ADC->ADCR &= ~(7 << ADCR_START);
70
71 return (LPC_ADC->ADGDR >> 4) & 0xFFF;
72}
