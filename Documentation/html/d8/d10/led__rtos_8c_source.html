25static void LED_RTOS_Task(void *pvParameters) {
26 LED_RTOS_Command *cmd;
27 while (true) {
28 if (xQueueReceive(xLedQueue, &cmd, portMAX_DELAY) == pdTRUE) {
29 switch (cmd->command) {
30 case LED_CMD_ON:
31 LED_On(cmd->color, cmd->model);
32 break;
33 case LED_CMD_OFF:
34 LED_Off(cmd->color, cmd->model);
35 break;
36 case LED_CMD_TOGGLE:
37 LED_Toggle(cmd->color, cmd->model);
38 break;
39 }
40 vPortFree(cmd);
41 }
42 }
43}

45static void LED_RTOS_Init( void ) {
46 if (xLedQueue == NULL){
47 xLedQueue = xQueueCreate(10, sizeof(LED_RTOS_Command *));
48 vQueueAddToRegistry(xLedQueue , "LQR");
49 }
50
51 if (xLedTaskHandle == NULL){
52 xTaskCreate(LED_RTOS_Task, "LED_RTOS_Task",
53 configMINIMAL_STACK_SIZE, NULL, 1, &xLedTaskHandle);
54 }
55
56 if (xLedMutex == NULL){
57 xLedMutex = xSemaphoreCreateMutex();
58 }
59}

96bool LED_RTOS_GetState(LED_ColorValueType color, bool *ledState) {
97 if (ledState == NULL || xLedMutex == NULL) return false;
98
99 for (int mutexRetry = 0; mutexRetry < LED_MAX_MUTEX_RETRY; ++mutexRetry) {
100 if (xSemaphoreTake(xLedMutex, LED_GETSTATE_TIMEOUT) == pdTRUE) {
101 *ledState = LED_GetState(color);
102 xSemaphoreGive(xLedMutex);
103 return true;
104 }
105 vTaskDelay(LED_GETSTATE_DELAY);
106 }
107
108 return false;
109}

111void LED_RTOS_On(LED_ColorValueType color, LED_ModelType model) {
112 if (!LED_RTOS_Ready()) return;
113 LED_RTOS_Command *cmd = LED_RTOS_AllocCommand();
114 if (cmd == NULL) return;
115 cmd->color = color;
116 cmd->command = LED_CMD_ON;
117 cmd->model = model;
118 if (xQueueSend(xLedQueue, &cmd, portMAX_DELAY) != pdTRUE) {
119 vPortFree(cmd);
120 }
121}
